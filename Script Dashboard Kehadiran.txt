<script>
const API = "https://script.google.com/macros/s/AKfycbwrG9e4hirDV97y0ZEVJHv-yYtt14g4eaYUqOZ_3wj4uV9I4eo6pDfmrf0AGhi_ZiJrJw/exec";

let currentPage = 1;
const rowsPerPage = 10;
let allData = [];
let chart;

// 🔥 LOAD DATA DARI API
fetch(API)
.then(res => res.json())
.then(res => {

  allData = res.data;

  let s = res.summary;

  document.getElementById("total").innerText = s.total;
  document.getElementById("hadir").innerText = s.hadir;
  document.getElementById("tidak").innerText = s.tidak;
  document.getElementById("persen").innerText = s.persen+"%";

  renderChart(s.hadir, s.tidak);
  renderTable();
});

// 🔥 TABLE
function renderTable(){
  let keyword = document.getElementById("search").value.toLowerCase();

  let filtered = allData.filter(x => 
    x.nama.toLowerCase().includes(keyword)
  );

  let start = (currentPage - 1) * rowsPerPage;
  let paginatedData = filtered.slice(start, start + rowsPerPage);

  let html = "";

  if(paginatedData.length === 0){
    html = "<tr><td colspan='3'>Data tidak ditemukan</td></tr>";
  } else {
    paginatedData.forEach((d,i)=>{
      html += `
        <tr>
          <td>${start + i + 1}</td>
          <td>${d.nama}</td>
          <td class="${d.kehadiran==='Hadir'?'status-hadir':'status-tidak'}">
            ${d.kehadiran}
          </td>
        </tr>
      `;
    });
  }

  document.getElementById("data").innerHTML = html;
  renderPagination(filtered.length);
}

// 🔥 PAGINATION
function renderPagination(total){
  let totalPages = Math.ceil(total / rowsPerPage);
  let html = "";

  for(let i=1;i<=totalPages;i++){
    html += `<button onclick="changePage(${i})">${i}</button>`;
  }

  document.getElementById("pagination").innerHTML = html;
}

function changePage(p){
  currentPage = p;
  renderTable();
}

// 🔥 SEARCH
document.getElementById("search").addEventListener("input", ()=>{
  currentPage = 1;
  renderTable();
});

// 🔥 CHART
function renderChart(hadir, tidak){
  const ctx = document.getElementById('chartKehadiran');

  chart = new Chart(ctx, {
    type: 'doughnut',
    data: {
      labels: ['Hadir','Tidak Hadir'],
      datasets: [{
        data: [hadir, tidak],
        backgroundColor: ['#198754','#dc3545']
      }]
    }
  });
}
